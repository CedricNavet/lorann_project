425     /**
426      * Generate Collision for item, sprite, monster spaceShip
427      */
428     public void checkCollisions() {
429 
430         Rectangle sp = spaceShip.getBounds();
431         Rectangle dopen = door_open.getBounds();
432         Rectangle dclose = door_close.getBounds();
433         Rectangle crystale = crystal.getBounds();
434         
435         for(Sprite sprites: sprite) {
436         	Rectangle wall = sprites.getBounds();
437         	
438         	if (sp.intersects(wall)) {
439              	System.out.println("Ca touche");
440              		spaceShip.reversemove();
441             }
442         	
443         	if(dclose.intersects(sp)) {
444         		spaceShip.reversemove();
445         		isgameover=true;
446         	}
447         	
448         	if(crystale.intersects(sp)) {
449         		iscrystaltake=true;
450         		crystal.setVisible(false);
451         		crystal.x=-400;
452         		crystal.y=-400;
453         		System.out.println("tocuhe le crys");
454         	}
455         	
456         	if(iscrystaltake==true && dopen.intersects(sp)) {
457         		iswin=true;
458         	}
459         	
460             for(Sprite items : item) {
461             	Rectangle r_item = items.getBounds();
462             	if(sp.intersects(r_item)) {
463             		items.x=4000;
464             		items.y=4000;
465             		System.out.println("ca touche l'item");
466             	}
467             }
468             for(Monster monsters: monster) {
469             	Rectangle monsters1= monsters.getBounds();
470             	
471             	if (sp.intersects(monsters1)) {
472                  	System.out.println("Ca touche le monstre");
473                  	isgameover =true;
474                 }
475             	if (wall.intersects(monsters1)||dopen.intersects(monsters1)||dclose.intersects(monsters1)) {
476             		monsters.reversMouvement();
477             	}
478 
479                 List<Missile> ms =spaceShip.getMissiles();
480                 
481                 for(Missile missile : ms) {
482                 	Rectangle miss = missile.getBounds();
483                 	 if(wall.intersects(miss)) {
484                       	missile.setVisible(false);                      
485                       }
486                 	 if(miss.intersects(monsters1)) {
487                 		 monsters.x=-400;
488                 		 monsters.y=-400;
489                 		 System.out.println("touchééé");
490                 	 }
491                 }     
492             }    
493         }
